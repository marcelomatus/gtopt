% ========== Biographies ==========
\begin{IEEEbiography}{Marcelo Matus~A.}
  received the Electrical Engineer degree from Pontificia Universidad
  Cat\'olica de Chile and the Ph.D. degree in electrical and computer
  engineering from the University of Arizona, Tucson, AZ, USA.  He is
  currently a Senior Researcher at Black Bear Engineering SpA and at
  the Centro Nacional de Inteligencia Artificial (CENIA), Santiago,
  Chile.  He was formerly Subdirector of the Centre for Energy (Centro
  de Energ\'ia), Universidad de Chile.  His research interests include
  power system optimization, mathematical programming, and
  high-performance computing applied to energy systems.  He is the
  lead developer of \gtopt{} and the \fesop{} framework.
\end{IEEEbiography}

\begin{IEEEbiography}{Mar\'ia del Pilar Buitrago Villada}
  received the B.Sc. and M.Sc. degrees in electrical engineering from
  Universidad Nacional de Colombia, Medell\'in, and the Ph.D. degree in
  engineering from the same university.  Her research focuses on power
  system expansion planning, renewable energy integration, and
  optimization under uncertainty.  She is a co-developer of the \fesop{}
  framework and has contributed to its application in long-term planning
  of the Colombian and Chilean power systems.
\end{IEEEbiography}

\begin{IEEEbiography}{Erick Sierra Baeza}
  received the Civil Electrical Engineer degree from Universidad de Chile.
  He is currently with Black Bear Engineering SpA, Santiago, Chile, and
  affiliated with CENIA.  He previously worked at the Centre for Energy,
  Universidad de Chile.  His professional experience spans wind energy
  integration, capacity expansion planning, hydrological modeling, and
  development of optimization tools for energy market analysis.
\end{IEEEbiography}

\begin{IEEEbiography}{Carlos Enrique Garc\'ia Bujanda}
  is a Researcher at Black Bear Engineering SpA and affiliated with the
  Department of Informatics, Universidad de Santiago de Chile.  His
  research interests include power systems optimization, renewable energy
  integration planning, and development of computational tools for
  long-term electricity system expansion.  He is a co-developer of the
  \fesop{} framework.
\end{IEEEbiography}

\begin{IEEEbiography}{Marcos Falcone Salazar}
  received the Electrical Engineer and M.Sc. degrees in electrical
  engineering from Universidad de Chile and the MBA degree from
  Universidad Adolfo Ib\'a\~nez, Santiago, Chile.  He held management
  positions at Colb\'un~S.A. including Sub-manager of Planning and
  Development, where he directed an R\&D initiative between Colb\'un,
  Universidad de Chile, Fondecyt, and CORFO that led to the development
  of a novel electricity market planning and operation model.  He is
  currently a Professor at the School of Engineering, Universidad Adolfo
  Ib\'a\~nez.  His professional interests include power system expansion
  planning, energy project valuation, and electricity market design.
\end{IEEEbiography}

\begin{IEEEbiography}{Gonzalo Bustos Turu}
  received the B.Sc. and M.Sc. degrees in electrical engineering from
  Universidad de Chile in 2009, the M.Sc. degree in sustainable energy
  futures from Imperial College London in 2013, and the Ph.D. degree
  from Imperial College London in 2018, working on smart urban energy
  systems under the supervision of Prof.~Nilay Shah.  He is currently
  a Research Associate at Imperial College London and affiliated with
  the Centre for Energy, Universidad de Chile.  His research interests
  include urban energy system modeling, demand response, electric vehicle
  integration, and district-level predictive control.
\end{IEEEbiography}
