\section{Mathematical Formulation}
\label{sec:formulation}

\subsection{Sets and Indices}

The GTEP problem is defined over the following sets:

\begin{itemize}
  \item $\setS$: scenarios, indexed by $s$, with probability $\pi_s$
  \item $\setT$: stages (planning periods), indexed by $t$
  \item $\setB_t$: blocks (operating periods) in stage $t$, indexed by $b$,
        with duration $\Delta_b$ [h]
  \item $\setN$: buses (electrical nodes), indexed by $n$
  \item $\setG$: generators, indexed by $g$, with bus $n_g$
  \item $\setD$: demands, indexed by $d$, with bus $n_d$
  \item $\setL$: transmission lines, indexed by $\ell$, connecting buses
        $a_\ell$ and $b_\ell$
  \item $\setE$: batteries (energy storage), indexed by $e$
  \item $\mathcal{V}$: converters (battery$\leftrightarrow$grid), indexed by $v$
  \item $\setJ$: hydraulic junctions, indexed by $j$
  \item $\setR$: reservoirs, indexed by $r$, with junction $j_r$
  \item $\setW$: waterways, indexed by $w$
  \item $\setH$: turbines, indexed by $h$
  \item $\setZ$: reserve zones, indexed by $z$
  \item $\mathcal{P}$: reserve provisions, indexed by $p$
  \item $\setFR$: flow rights (flow-based water rights), indexed by $i$
  \item $\setVR$: volume rights (volume-based water rights), indexed by $o$
  \item $\setHP$: hydraulic pumps, indexed by $h$
  \item $\setLNG$: LNG storage terminals, indexed by $\tau$
  \item $\setUC$: generators with unit commitment, indexed by $g$
  \item $\setIZ$: inertia zones, indexed by $z$
  \item $\setIP$: inertia provisions, indexed by $p$
\end{itemize}

\subsection{Decision Variables}

\noindent\textbf{Operational variables} (per scenario $s$, stage $t$, block $b$):
\begin{align}
  p_{g,s,t,b}  &\in [\underline{p}_g, \overline{p}_g]
    & \text{generator output [MW]} \label{eq:pgen} \\
  \ell_{d,s,t,b} &\in [0, \overline{\ell}_d]
    & \text{served demand [MW]} \label{eq:load} \\
  q_{d,s,t,b}  &\geq 0
    & \text{unserved demand [MW]} \label{eq:fail} \\
  f_{\ell,s,t,b} &\in [-\overline{F}^{ba}_\ell, \overline{F}^{ab}_\ell]
    & \text{line flow [MW]} \label{eq:flow} \\
  \theta_{n,s,t,b} &\in \R
    & \text{voltage angle [rad]} \label{eq:theta}
\end{align}

\noindent\textbf{Storage state variables}:
\begin{align}
  e_{e,s,t,b}  &\in [\underline{e}_e, \overline{e}_e]
    & \text{battery SoC [MWh]} \label{eq:soc} \\
  v_{r,s,t,b}  &\in [\underline{v}_r, \overline{v}_r]
    & \text{reservoir volume [hm\textsuperscript{3}]} \label{eq:vol} \\
  e_{\tau,s,t,b}^{\text{lng}}
               &\in [\underline{V}_\tau, \overline{V}_\tau]
    & \text{LNG tank volume [m\textsuperscript{3}]} \label{eq:lng_vol}
\end{align}

\noindent\textbf{Unit commitment variables} (per scenario $s$, stage $t$, period $p$):
\begin{align}
  u_{g,s,t,p} &\in \{0,1\} & \text{unit status (1 = online)} \\
  v_{g,s,t,p} &\in \{0,1\} & \text{startup event} \\
  w_{g,s,t,p} &\in \{0,1\} & \text{shutdown event}
\end{align}

\noindent\textbf{Investment variables} (per stage $t$):
\begin{equation}
  m_{g,t} \in [0, \overline{m}_g]
  \quad \text{expansion modules built} \label{eq:expmod}
\end{equation}

\subsection{Objective Function}

Minimize total expected discounted cost:
\begin{equation}
\label{eq:objective}
\min \sum_{s \in \setS} \pi_s \sum_{t \in \setT} \delta_t
\left[
  \sum_{b \in \setB_t} \Delta_b \cdot C^\text{op}_{s,t,b}
  + C^\text{inv}_t
\right]
\end{equation}
where $\delta_t = (1+r)^{-\tau_t/8760}$ is the discount factor with annual
rate $r$ and $\tau_t$ the cumulative hours to stage $t$.

The operational cost for each $(s,t,b)$ is:
\begin{multline}
\label{eq:opcost}
C^\text{op}_{s,t,b} = \sum_{g \in \setG} c_g \, p_{g,s,t,b}
  + \sum_{d \in \setD} c^f_d \, q_{d,s,t,b} \\
  + \sum_{\ell \in \setL} c_\ell
    (f^+_{\ell,s,t,b} + f^-_{\ell,s,t,b})
\end{multline}

The investment cost for stage $t$ is:
\begin{multline}
\label{eq:invcost}
C^\text{inv}_t = \sum_{g \in \setG} \omega^\text{cap}_{g,t} m_{g,t}
  + \sum_{d \in \setD} \omega^\text{cap}_{d,t} m_{d,t} \\
  + \sum_{\ell \in \setL} \omega^\text{cap}_{\ell,t} m_{\ell,t}
  + \sum_{e \in \setE} \omega^\text{cap}_{e,t} m_{e,t}
\end{multline}
where $\omega^\text{cap}_{(\cdot),t} = K_{(\cdot)} \delta_t T_t /
\sigma$ is the weighted investment cost coefficient,
$c_g$ is the variable generation cost [\$/MWh], $c^f_d$ is the
demand failure penalty [\$/MWh], $c_\ell$ is the transmission cost
[\$/MWh], $K_{(\cdot)}$ is the annualized investment cost
[\$/year] per module for each component, and $T_t$ is the stage
duration [h].  All objective coefficients are divided by a
scaling factor $\sigma$ (\texttt{scale\_objective}, default 1000) to
improve solver numerical conditioning.  Concretely, each operational LP
cost coefficient is weighted by $\omega_{s,t,b} = \pi_s \delta_t
\Delta_b / \sigma$, so the reported objective equals $z/\sigma$.

\subsection{Bus Power Balance}
\label{sec:balance}
\begin{multline}
\label{eq:balance}
\sum_{\substack{g \in \setG \\ n_g = n}} (1{-}\lambda_g) p_{g,s,t,b}
- \sum_{\substack{d \in \setD \\ n_d = n}}
  (1{+}\lambda_d)\,\ell_{d,s,t,b} \\
+ \sum_{\ell \in \setL} A_{n,\ell} \, f_{\ell,s,t,b} = 0
\end{multline}
where $\lambda_g, \lambda_d$ are injection and withdrawal loss factors,
respectively.  The line flow terms use directional contributions rather
than a simple incidence matrix: for each line~$\ell$ from bus~$a_\ell$
to~$b_\ell$, the sending bus sees $-f^+_\ell$ (or $+(1-\lambda_\ell)
f^-_\ell$ when losses are modeled) and the receiving bus sees
$+(1-\lambda_\ell)f^+_\ell$ (or $-f^-_\ell$), where $\lambda_\ell$
is the line loss factor.  When $\lambda_\ell=0$, a single bidirectional
variable suffices and the incidence reduces to $A_{n,\ell} \in
\{+1,-1,0\}$.  The dual of~\eqref{eq:balance} is the locational marginal
price (LMP) at bus~$n$.

\subsection{Demand Balance}

When demand curtailment is allowed, served and unserved demand sum to the
scheduled load:
\begin{equation}
\label{eq:demand_balance}
\ell_{d,s,t,b} + q_{d,s,t,b} = \overline{\ell}_{d,t,b}
\quad \forall\; d, s, t, b
\end{equation}
where $\overline{\ell}_{d,t,b}$ is the scheduled demand at block~$b$,
incorporating any demand profile scaling.  When curtailment is disabled,
the load is fixed: $\ell_{d,s,t,b} = \overline{\ell}_{d,t,b}$.

\subsection{DC Power Flow (Kirchhoff Constraints)}

When \texttt{use\_kirchhoff} is enabled, for each line $\ell$ and
$(s,t,b)$:
\begin{equation}
\label{eq:kirchhoff}
f_{\ell,s,t,b} = \frac{V_\ell^2}{X_\ell}
  \left(\theta_{a_\ell,s,t,b} - \theta_{b_\ell,s,t,b}\right)
\end{equation}
where $X_\ell$ is the line reactance [$\Omega$] and $V_\ell$ is the
nominal voltage [kV].  A scaling factor $\sigma_\theta$
(\texttt{scale\_theta}, default 1000) is applied to the angle variables
for numerical conditioning: $\theta^\text{scaled} = \sigma_\theta \cdot
\theta$.

Line flow is bounded by transfer capacities:
\begin{equation}
-\overline{F}^{ba}_\ell \leq f_{\ell,s,t,b}
  \leq \overline{F}^{ab}_\ell
\end{equation}

\paragraph{Transmission losses.}
When line losses are enabled ($\lambda_\ell > 0$), separate forward and
reverse flow variables $f^+_\ell, f^-_\ell \geq 0$ replace the
single bidirectional variable.  The receiving bus sees only
$(1-\lambda_\ell)$ of the dispatched power, so losses are allocated at
the receiving end.  The Kirchhoff constraint~\eqref{eq:kirchhoff} then
uses $f^+_\ell - f^-_\ell$ in place of the single~$f_\ell$.

\paragraph{Transformer model.}
For tap-changing and phase-shifting transformers, the Kirchhoff
constraint generalizes to:
\begin{equation}
\label{eq:transformer}
-\theta_{a_\ell} + \theta_{b_\ell}
  + (\tau_\ell \chi_\ell)(f^+_\ell - f^-_\ell)
  = -\sigma_\theta \, \psi_{\ell}
\end{equation}
where $\tau_\ell$ is the off-nominal tap ratio (default~1),
$\chi_\ell = \sigma_\theta X_\ell / V_\ell^2$ is the scaled reactance,
and $\psi_\ell$ is the phase-shift angle [rad].
When $\tau_\ell = 1$ and $\psi_\ell = 0$,~\eqref{eq:transformer}
reduces to~\eqref{eq:kirchhoff}.

\subsection{Battery Energy Storage}
\label{sec:battery}

State-of-charge balance for battery $e$ between consecutive blocks:
\begin{multline}
\label{eq:soc_balance}
e_{e,s,t,b} = e_{e,s,t,b-1}(1 - \mu_e^h \Delta_b) \\
  + p^\text{in}_{e,s,t,b} \cdot \eta^\text{in}_e \cdot \Delta_b
  - p^\text{out}_{e,s,t,b} \cdot \Delta_b / \eta^\text{out}_e
\end{multline}
where $\eta^\text{in}_e$ and $\eta^\text{out}_e$ are charge and discharge
efficiencies, $p^\text{in}, p^\text{out}$ are charge and discharge
power [MW], and $\mu_e^h = \mu_e / 8760$ is the hourly self-discharge
rate derived from the annual loss rate~$\mu_e$.
Initial SoC $e_{e,\cdot,\cdot,0} = e^\text{ini}_e$ is an
equality constraint in the first stage; the final SoC of stage~$t$
links to the initial SoC of stage~$t{+}1$.

\subsection{Converter Constraints}

A converter~$v$ couples a battery~$e$ to the electrical network through
a generator~$g$ (discharge path) and a demand~$d$ (charge path) via a
power conversion rate~$\rho_v$:
\begin{align}
\label{eq:conv_discharge}
p_{g,s,t,b} &= \rho_v \cdot p^\text{out}_{e,s,t,b} \\
\label{eq:conv_charge}
\ell_{d,s,t,b} &= \rho_v \cdot p^\text{in}_{e,s,t,b}
\end{align}
When the converter has a capacity limit~$\overline{C}_v$, a joint bound
applies: $p_{g,s,t,b} + \ell_{d,s,t,b} \leq \bar{C}_{v,t}$.
Converter capacity is expandable using the same modular framework
as~\eqref{eq:capacity}.

\subsection{Hydro Cascade}
\label{sec:hydro}

For each junction $j$ with reservoir $r$, the water balance is:
\begin{multline}
\label{eq:hydro_balance}
v_{r,s,t,b} = v_{r,s,t,b-1} \\
  + \alpha_r \Delta_b \!\Bigl(
    \textstyle\sum_{w \in \text{in}(j)} q_{w}
    - \textstyle\sum_{w \in \text{out}(j)} q_{w}
    + F_j
  \Bigr)
\end{multline}
where $q_w$ is the waterway discharge [m$^3$/s], $F_j$ is the exogenous
inflow at junction $j$, and $\alpha_r = 0.0036$ converts flow rates to
volume increments: since 1~m$^3$/s flowing for 1~hour equals
3600~m$^3$ = 0.0036~hm$^3$, we have $\alpha_r = 3.6 \times 10^{-3}$
[hm$^3$/(m$^3$/s$\cdot$h)].

Each turbine $h$ linking waterway $w$ to generator $g$ provides:
\begin{equation}
\label{eq:turbine}
p_{g,s,t,b} \leq \rho_h \cdot q_{w,s,t,b}
\end{equation}
where $\rho_h$ is the water-to-power conversion rate [MW$\cdot$s/m$^3$].

\paragraph{Volume-dependent turbine efficiency.}
In practice, the conversion rate depends on the hydraulic head, which
varies with reservoir volume.  When defined, $\rho_h$ is replaced by a
concave piecewise-linear function:
\begin{equation}
\label{eq:vol_efficiency}
\rho_h(v) = \min_{i=1}^{N}
  \bigl\{ c_i + m_i (v - v_i) \bigr\}
\end{equation}
where $(v_i, m_i, c_i)$ are volume breakpoints, slopes, and intercepts
of~$N$ segments.  In the SDDP method, the active segment is updated at
each forward-pass iteration using the current reservoir volume.

\paragraph{Filtration (seepage).}
Seepage from a waterway into a reservoir is modeled as a linear function
of the average volume:
\begin{equation}
\label{eq:filtration}
\varphi_{i,s,t,b} = a_i + b_i \cdot
  \frac{v_{r,s,t,b-1} + v_{r,s,t,b}}{2}
\end{equation}
where $a_i$ is a constant term and $b_i$ is the seepage rate per unit
volume.  As with turbine efficiency, this admits a piecewise-linear
extension selected by the last known volume.

\subsection{Capacity Expansion}

Installed capacity evolves as:
\begin{equation}
\label{eq:capacity}
\bar{C}_{g,t} = \bar{C}_{g,t-1}(1-\xi_{g,t}) + M_g \cdot m_{g,t}
\end{equation}
where $\bar{C}_{g,0}$ is the initial installed capacity, $\xi_{g,t}$ is
the stage derating factor, $M_g$ is the capacity per expansion module
(\texttt{expcap}), and $m_{g,t}$ is the number of modules built.  The
generator output is bounded by the installed capacity:
\begin{equation}
p_{g,s,t,b} \leq \bar{C}_{g,t}
\end{equation}
Analogous expansion constraints apply to demands, lines, batteries, and
converters.

\subsection{Generator Profile Constraints}

When a generator has a capacity profile $\phi_g(s,t,b)$ (e.g., solar
irradiance or wind availability), the output is forced to follow the
profile:
\begin{equation}
\label{eq:profile}
p_{g,s,t,b} + s_{g,s,t,b} = \phi_g(s,t,b) \cdot \bar{C}_{g,t}
\end{equation}
where $\phi_g(s,t,b) \in [0,1]$ is the per-block capacity factor and
$s_{g,s,t,b} \geq 0$ is the curtailed (spilled) generation.  The
spillover variable carries cost $c_g^\text{spill} \cdot \omega_{s,t,b}$,
which may be set to zero to allow free curtailment of renewables.  This
lets the optimizer spill excess renewable production when it is not
economically efficient to deliver.

\subsection{Demand Minimum Energy}

An optional minimum energy constraint ensures that a demand~$d$ is
served a minimum total energy within each stage:
\begin{equation}
\label{eq:min_energy}
\sum_{b \in \setB_t} \ell^\text{man}_{d,s,t,b} \cdot \Delta_b
  \geq E_d^\text{min}
\end{equation}
where $\ell^\text{man}_{d,s,t,b}$ is the mandatory served load and
$E_d^\text{min}$ [MWh] is the stage energy requirement.

\subsection{Reserve Constraints}
\label{sec:reserve}

For each reserve zone $z$ and $(s,t,b)$:
\begin{equation}
\label{eq:reserve_up}
\sum_{p \in \mathcal{P}_z} \gamma_{p,t} \, r^\text{up}_{p,s,t,b}
  + q^\text{up}_{z,s,t,b}
  \geq R^\text{up}_{z,s,t,b}
\end{equation}
where $r^\text{up}_{p,s,t,b}$ is the up-reserve provision from
provision~$p$, $\gamma_{p,t}$ is the provision factor (typically~1),
$q^\text{up}_{z,s,t,b}$ is unserved reserve, and
$R^\text{up}_{z,s,t,b}$ is the zone requirement.  A symmetric constraint
exists for down-reserve.  Reserve provision is bounded by the generator's
headroom: $p_{g,s,t,b} + r^\text{up}_{p,s,t,b} \leq \bar{C}_{g,t}$.

\subsection{Hydro Pump}
\label{sec:pump}

A \texttt{Pump} element $h \in \setHP$ models a hydraulic pump
that consumes electrical power from a demand $d$ and raises water
through a waterway $w$ from a downstream junction to an upstream
junction---the reverse of a turbine.  The conversion constraint ties
the electrical load to the pumped flow:
\begin{equation}
\label{eq:pump}
\frac{\phi_{h,t}}{\eta_{h,t}} \, \varphi_{w,s,t,b} \leq \ell_{d,s,t,b}
\qquad \forall\, h \in \setHP,\; s,t,b
\end{equation}
where $\phi_{h,t}$ [MW/(m\textsuperscript{3}/s)] is the pump factor,
$\eta_{h,t} \in (0,1]$ is the stage efficiency, $\varphi_{w,s,t,b}$
[m\textsuperscript{3}/s] is the waterway flow, and
$\ell_{d,s,t,b}$ [MW] is the served electrical load.  An optional
capacity constraint bounds the flow:
$\varphi_{w,s,t,b} \leq \overline{Q}_{h,t}$.

The pump element only adds coupling rows to the LP; the waterway
retains its junction water balance contribution (Section~\ref{sec:hydro})
and the demand retains its bus power balance contribution
(Section~\ref{sec:balance}).  Reversible turbine-pump units (e.g.
pump-storage hydro) are modeled as a pair of elements: a
\texttt{Turbine} on the generation waterway and a \texttt{Pump} on the
pumping waterway.  Economic optimality (round-trip efficiency
$< 1.0$) prevents simultaneous generation and pumping without
explicit binary exclusivity constraints.

\subsection{LNG Terminal}
\label{sec:lng}

An \texttt{LngTerminal} element $\tau \in \setLNG$ models an
LNG storage tank whose inventory is coupled to gas-fired generators
through their heat rates.  The tank volume balance mirrors the battery
SoC equation (Section~\ref{sec:battery}) with delivery inflow and
boil-off:
\begin{equation}
\label{eq:lng_balance}
e_{\tau,s,t,b}^{\text{lng}}
  = e_{\tau,s,t,b-1}^{\text{lng}} \bigl(1 - \lambda_\tau^h \Delta_b\bigr)
  + \rho_\tau \Delta_b \, d_{\tau,s,t,b}^{\text{lng}}
  - \rho_\tau \Delta_b \, \text{vent}_{\tau,s,t,b}
  - \rho_\tau \Delta_b
      \sum_{g \in \mathcal{G}_\tau} \eta_g^{hr} \, p_{g,s,t,b}
\end{equation}
where $e_{\tau,s,t,b}^{\text{lng}}$ [m\textsuperscript{3}] is the tank
volume, $\lambda_\tau^h = \lambda_\tau / 8760$ is the per-hour
boil-off rate, $\rho_\tau$ is the unit-conversion factor
(m\textsuperscript{3}/(m\textsuperscript{3}/h$\cdot$h), default~1),
$d_{\tau,s,t,b}^{\text{lng}}$ is the delivery inflow, $\text{vent}$
is a vented-gas variable penalized at `spillway\_cost', and
$\eta_g^{hr}$ [m\textsuperscript{3}/MWh] is the heat rate of each
linked generator $g \in \mathcal{G}_\tau$.

The last term is implemented by injecting a coefficient
$-\rho_\tau \eta_g^{hr} \Delta_b$ onto each generator's generation
variable in the existing tank balance row.  Volume bounds and
initial/final conditions follow the generic storage template.  The
delivery schedule is fixed to a constant hourly rate
$d_{\tau,s,t,b} = D_{\tau,t} / T_t$ across each stage.

This formulation enables the optimizer to manage fuel inventory (buy
decisions and storage costs) jointly with economic dispatch, capturing
supply security constraints and seasonal LNG price dynamics.

\subsection{Unit Commitment}
\label{sec:uc}

\gtopt{} supports thermal unit commitment through two elements.  The
full three-binary \texttt{Commitment} element uses the classical
three-bin (3-bin) formulation with status~$u$, startup~$v$, and
shutdown~$w$ binary variables~\cite{morales2013}:
\begin{align}
u_{g,s,t,p} - u_{g,s,t,p-1} - v_{g,s,t,p} + w_{g,s,t,p}
  &= 0
  \label{eq:uc_logic} \\
v_{g,s,t,p} + w_{g,s,t,p}
  &\leq 1
  \label{eq:uc_excl} \\
\underline{P}_g u_{g,s,t,p(b)}
  \leq p_{g,s,t,b}
  &\leq \overline{P}_g u_{g,s,t,p(b)}
  \label{eq:uc_bounds}
\end{align}
Minimum up/down time constraints, ramp limits, hot/warm/cold startup
cost tiers, and piecewise heat-rate curves are all supported.  Binary
variables live on a commitment-period grid that can be coarser than the
dispatch blocks, enabling sub-hourly dispatch with hourly on/off
decisions.

The lighter \texttt{SimpleCommitment} attaches one binary per block
without startup/shutdown bookkeeping:
\begin{equation}
\label{eq:sc_bounds}
\underline{P}_g u_{g,s,t,b}^{\text{sc}}
  \leq p_{g,s,t,b}
  \leq \overline{P}_g u_{g,s,t,b}^{\text{sc}}
\end{equation}
This is sufficient for daily unit commitment in many hydrothermal systems
and for inertia modeling where only the online/offline status matters.

Commitment constraints apply only to stages flagged
\texttt{chronological}~$=$ true (consecutive hourly blocks).  On
representative-period stages (duration-weighted blocks not in calendar
order), binary variables are silently relaxed to continuous.

\subsection{CO\textsubscript{2} Emission Pricing and Caps}
\label{sec:emission}

gtopt supports two mechanisms for carbon pricing.  A system-wide emission
price $c^{\text{em}}$ [\$/tCO\textsubscript{2}] applied to generators
with emission intensity $f_g$ [tCO\textsubscript{2}/MWh] augments each
generator's variable cost:
\begin{equation}
\label{eq:emission_cost}
c_g^{\text{eff}} = c_g + c^{\text{em}} \cdot f_g
\end{equation}
A per-stage emission cap $\overline{E}^{\text{em}}_t$
[tCO\textsubscript{2}/year] adds an aggregate cap constraint whose
dual variable yields the endogenous carbon price:
\begin{equation}
\label{eq:emission_cap}
\sum_{g,s,b} \pi_s \cdot f_g \cdot p_{g,s,t,b} \cdot \Delta_b
  \leq \overline{E}^{\text{em}}_t
\end{equation}
For the piecewise heat-rate model (Section~\ref{sec:uc}), the per-segment
emission factor \texttt{fuel\_emission\_factor} $\times$ \texttt{heat\_rate}
replaces $f_g$.

\subsection{Synchronous Inertia Requirements}
\label{sec:inertia}

As variable renewable penetration increases, maintaining sufficient
synchronous inertia for frequency stability becomes a binding operational
constraint.  \gtopt{} models inertia requirements through two elements
mirroring the reserve-zone formulation (Section~\ref{sec:reserve}).

An \texttt{InertiaZone} $z \in \setIZ$ specifies a minimum inertia
requirement $\overline{R}_{z,t,b}^{H}$ [MWs]:
\begin{equation}
\label{eq:inertia_req}
\sum_{p \in \mathcal{P}_z} \Phi_{p,t} \, r_{p,s,t,b}^{H}
  \geq \overline{R}_{z,t,b}^{H}
\end{equation}
An \texttt{InertiaProvision} $p \in \setIP$ links a generator to
one or more zones.  The provision variable
$r_{p,s,t,b}^{H} \in [0, \overline{R}_{p,t,b}^{H}]$ [MW] is bounded
by a downward coupling row:
\begin{equation}
\label{eq:inertia_coupling}
p_{g,s,t,b} \geq r_{p,s,t,b}^{H}
\end{equation}
The conversion factor from MW dispatch to MWs of inertia is:
\begin{equation}
\Phi_{p,t} = \frac{H_p \cdot S_p}{\underline{P}_g}
\end{equation}
where $H_p$ [s] is the machine inertia constant and $S_p$ [MVA] is the
rated apparent power.  A user-specified \texttt{provision\_factor} may
override this formula.  Shortage is penalized at the zone's
\texttt{cost} [\$/MWs], enabling the optimizer to trade off committed
generation against inertia value.

\subsection{Water Rights and Irrigation Agreements}
\label{sec:rights}

Hydro-dominated power systems often operate under legally-binding
irrigation agreements that partition turbine discharge into distinct
rights categories---irrigation, electrical, mixed---with
volume-dependent allocations that vary with reservoir level.  In Chile,
the Laja Agreement (1958, updated 2017) and the Maule Convention (1947)
govern two of the largest hydro cascades in the national interconnected
system, imposing seasonal flow obligations and multi-annual volume
accounting constraints on the optimizer.

Traditional tools such as PLP encode each agreement as a monolithic
Fortran module, tightly coupling the rights logic to the physical
topology.  \gtopt{} instead introduces two generic, data-driven
entities---\emph{FlowRight} and \emph{VolumeRight}---together with
the flexible \emph{UserConstraint} mechanism, forming a
\emph{rights-domain} layer parallel to the physical hydro topology.
This separation allows arbitrary irrigation agreements to be specified
declaratively in JSON and parameterized AMPL-inspired scripts without
modifying the solver code.

\subsubsection{Flow Right}

A flow right $i \in \setFR$ represents a flow-based water right
[m$^3$/s] that creates one LP column per block.  Two operating modes
are supported:
\begin{itemize}
  \item \emph{Fixed mode}: the extraction flow is fixed at a scheduled
    discharge value $\bar{\xi}_{i,s,t,b}$ (lower bound equals upper
    bound).
  \item \emph{Variable mode}: when a maximum flow $\overline{F}_i > 0$
    is specified, the optimizer decides the flow
    $\xi_{i,s,t,b} \in [0, \overline{F}_i]$.
\end{itemize}

When $\texttt{consumptive} = \texttt{true}$, the flow right subtracts
from the physical junction balance, coupling the rights domain to the
physical hydro topology.  An optional deficit variable with penalty
cost~$c_i^f$ ensures feasibility when the right cannot be fully served.

A stage-average hourly flow variable $\bar{\xi}^h_i$ is optionally
created to aggregate block-level flows into the duration-weighted
average required by the volume accumulation equations:
\begin{equation}
\label{eq:fr_avg}
\bar{\xi}^h_{i,s,t} = \sum_{b \in \setB_t}
  \frac{\Delta_b}{T_t} \, \xi_{i,s,t,b}
\end{equation}

\subsubsection{Volume Right}

A volume right $o \in \setVR$ is modeled as a virtual reservoir
(following the same \texttt{StorageLP} pattern as batteries) that
tracks accumulated volume [hm$^3$] across stages.  The volume balance
is:
\begin{equation}
\label{eq:vr_balance}
\nu_{o,s,t} = \nu_{o,s,t-1}
  - \alpha_o \sum_{b \in \setB_t} \Delta_b \, \xi^\text{in}_{o,s,t,b}
\end{equation}
where $\nu_{o,s,t}$ is the remaining volume of right~$o$ at the end of
stage~$t$, $\xi^\text{in}_{o,s,t,b}$ is the input extraction rate
[m$^3$/s], and $\alpha_o = 0.0036$~[hm$^3$/(m$^3$/s$\cdot$h)] is the
flow-to-volume conversion factor.

Volume rights support inter-stage state-variable coupling for SDDP,
optional demand satisfaction constraints
($\alpha_o \sum_b \Delta_b \xi^\text{in}_b + q^\text{fail} \geq D_o$),
and an automatic seasonal reset mechanism: when a
\texttt{reset\_month} is configured, the accumulated volume is reset to
the initial value at the start of each hydrological year.

\subsubsection{User Constraints and Parameters}
\label{sec:user_constraints}

The partition logic that links flow rights together---and the
proportional allocation rules of the Maule Convention---is expressed
through \emph{UserConstraints}: user-defined linear constraints written
in an AMPL-inspired pseudo-script.  Each constraint is a named linear
(in)equality over LP variables of any system element (generators,
demands, flow rights, volume rights, etc.) and can optionally restrict
its domain to specific scenarios, stages, or blocks.

For the Laja agreement, the total turbine flow (``gasto turbinado'')
must equal the sum of all partitioned extraction categories---irrigation
rights (``derechos de riego''), electrical rights (``derechos
el\'ectricos''), mixed rights (``derechos mixtos''), and anticipated
discharge (``gasto anticipado''):
\begin{equation}
\label{eq:uc_partition}
\xi^\text{gen} = \xi^\text{irr} + \xi^\text{elec}
  + \xi^\text{mix} + \xi^\text{antic}
\end{equation}
In the \texttt{.pampl} template this constraint reads:
\begin{verbatim}
constraint laja_particion_derechos
    "Flow partition":
  flow_right('laja_q_turbinado').flow =
      flow_right('laja_der_riego').flow
    + flow_right('laja_der_electrico').flow
    + flow_right('laja_der_mixto').flow
    + flow_right('laja_gasto_anticipado').flow;
\end{verbatim}

Entity names follow the PLP Spanish naming conventions (e.g.,
\texttt{laja\_der\_riego} for PLP's \texttt{IQDR}, ``Derecho de
Riego'') to maintain traceability with the reference Fortran
implementation.  This single UserConstraint replaces the hardcoded
balance node that traditional tools implement as a dedicated data
structure.

When a FlowRight has a positive \texttt{fail\_cost} and a nonzero
\texttt{discharge}, a deficit coupling constraint is automatically
generated:
\begin{equation}
\label{eq:deficit_coupling}
\xi_{i,b} + \delta_{i,b} \geq d_{i,b}
\end{equation}
where $\delta_{i,b}$ is the deficit variable penalized in the
objective.  This ensures the optimizer only uses the deficit
variable when the flow is genuinely constrained by junction
balance or dynamic bounds.

For the Maule Convention, the ordinary reserve zone enforces a
proportional split between electric ($p_e$) and irrigation ($p_r$)
extraction from the same shared flow:
\begin{align}
\label{eq:uc_maule_pct}
\xi^\text{elec}_\text{ord} &\leq
  p_e \bigl(\xi^\text{elec}_\text{ord} + \xi^\text{irr}_\text{ord}\bigr) \\
\xi^\text{irr}_\text{ord}  &\leq
  p_r \bigl(\xi^\text{elec}_\text{ord} + \xi^\text{irr}_\text{ord}\bigr)
\end{align}
where $p_e + p_r = 1$.  In \gtopt{}, these constraints reference the
Maule FlowRights using Spanish naming
(\texttt{maule\_gasto\_ordinario\_elec} for PLP's \texttt{IQMOE},
\texttt{maule\_gasto\_ordinario\_riego} for \texttt{IQMOR}),
replacing the hardcoded Fortran proportional-split logic in PLP's
\texttt{genpdmaulen.f}.

Named scalar or monthly-indexed \emph{UserParams} ($p_e$, $p_r$, cost
arrays, seasonal demand curves, etc.) are defined in the same script
file and resolved automatically from the stage calendar month, enabling
seasonally-varying constraints without per-stage hardcoding.
Each agreement is maintained in a separate \texttt{.pampl} file
(e.g., \texttt{laja\_agreement.pampl}, \texttt{maule\_agreement.pampl})
and loaded via the \texttt{user\_constraint\_files} array field in the
system JSON, keeping each agreement self-contained.
The \texttt{plp2gtopt} converter generates these templates automatically
from the parsed Fortran agreement parameters.

\subsubsection{Volume-Dependent Dynamic Bounds}

A key feature of irrigation agreements is that the available rights
depend on the current reservoir volume.  For example, the Laja agreement
defines four volume zones that progressively allocate rights as the
reservoir fills:
\begin{equation}
\label{eq:bound_rule}
\overline{F}_i(V) = \min\!\Bigl(
  c_k + m_k (V - V_k),\; \overline{F}_i^\text{cap}
\Bigr)
\end{equation}
where $(V_k, m_k, c_k)$ are breakpoints of a piecewise-linear function
and $V$ is the current volume of the source reservoir.  At each SDDP
iteration, the bound is re-evaluated from the forward-pass reservoir
volume and applied via \texttt{set\_col\_upp} on the corresponding flow
or extraction column.

\subsubsection{Design Advantages}

The rights-domain abstraction provides three advantages over the
monolithic approach used in traditional tools: (i)~new agreements are
specified declaratively in JSON and AMPL-inspired scripts without
modifying the solver code; (ii)~the same two generic entities
(FlowRight and VolumeRight) together with UserConstraints express both
the Laja and Maule agreements despite their very different structures
(4 volume zones vs.\ 3 reservoir portions, multi-annual vs.\ seasonal
reset, exact partition vs.\ proportional split); (iii)~volume-dependent
dynamic bounds integrate naturally with the SDDP forward-pass update
mechanism, reusing the same piecewise-linear evaluation used for
turbine efficiency.  The \texttt{plp2gtopt} converter automatically
transforms PLP's Fortran agreement parameters into the equivalent
\gtopt{} JSON configuration and \texttt{.pampl} template with verified
numerical equivalence.
